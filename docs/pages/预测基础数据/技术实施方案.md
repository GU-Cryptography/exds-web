## 功能概述

### 核心功能

这是一个**数据可用性查询和96点曲线对比分析**页面，允许用户:

1. **选择基准日期** (D日)
2. **查看数据可用性** - 历史10天到未来3天 (D-10到D+3) 的15个数据项的可用性状态
3. **多选数据项** - 跨数据类型、跨日期选择多个数据项
4. **96点曲线对比** - 在同一图表中对比多条负荷/出力预测曲线
5. **响应式设计** - 支持桌面端和移动端

### 数据项分类

#### 周预测数据 (5项) - 来源: `weekly_forecast` 集合
1. 周系统负荷预测
2. 周统调光伏预测
3. 周统调风电预测
4. 周水电(含抽蓄)预测
5. 次周省间联络线可用容量

#### 日前发布数据 (5项) - 来源: `daily_release` 集合
6. 短期系统负荷预测
7. 短期光伏预测
8. 短期风电预测
9. 非市场化机组出力预测
10. 联络线总计划

#### 实时发力数据 (5项) - 来源: `real_time_generation` + `real_time_tieline` 集合
11. 实际全网总出力
12. 实际风电出力
13. 实际光电出力
14. 实际水电(含抽蓄)出力
15. 联络线总计划值

```python
from pydantic import BaseModel, Field
from typing import List, Optional
from datetime import datetime

# ============ 数据点模型 ============
class TimeSeriesDataPoint(BaseModel):
    """时序数据点 - 96点中的一个点"""
    time: str = Field(..., description="业务时间标签: 00:15, 23:45, 24:00")
    value: float = Field(..., description="数值")
    timestamp: str = Field(..., description="ISO时间戳")

# ============ 数据可用性模型 ============
class DataAvailability(BaseModel):
    """数据可用性信息"""
    data_item_id: int = Field(..., description="数据项ID (1-15)")
    date: str = Field(..., description="日期 YYYY-MM-DD")
    is_available: bool = Field(..., description="是否有数据")
    sample_timestamp: Optional[str] = Field(None, description="样本时间戳(12:00点)")

class DataAvailabilityResponse(BaseModel):
    """数据可用性响应"""
    base_date: str = Field(..., description="基准日期")
    date_range: List[str] = Field(..., description="日期列表")
    availability_matrix: List[List[DataAvailability]] = Field(..., description="可用性矩阵")

# ============ 曲线数据模型 ============
class CurveDataRequest(BaseModel):
    """曲线数据请求"""
    data_item_id: int = Field(..., ge=1, le=15, description="数据项ID")
    date: str = Field(..., description="日期 YYYY-MM-DD")

class CurveDataResponse(BaseModel):
    """单条曲线数据响应"""
    data_item_id: int
    data_item_name: str = Field(..., description="数据项名称")
    date: str
    data: List[TimeSeriesDataPoint] = Field(..., description="96点数据")
    total_points: int
    completeness: float = Field(..., description="数据完整性百分比")

class MultipleCurvesResponse(BaseModel):
    """多曲线数据响应"""
    curves: List[CurveDataResponse]
```

```python
class ForecastBaseDataService:
    """预测基础数据服务"""

    def __init__(self, db: Database):
        self.db = db
        self._ensure_indexes()

    def _ensure_indexes(self):
        """确保索引存在(数据库已有索引,此处仅作验证)"""
        pass

    def get_data_availability(
        self,
        base_date: datetime,
        date_range_type: str = "recent_3"
    ) -> Dict[str, Any]:
        """
        获取数据可用性矩阵

        Args:
            base_date: 基准日期(D日)
            date_range_type: 日期范围类型
                - recent_3: D-2 到 D (3天)
                - recent_7: D-7 到 D (8天)
                - historical_10: D-10 到 D-1 (10天)

        Returns:
            数据可用性矩阵
        """
        # 1. 生成日期列表
        date_list = self._generate_date_list(base_date, date_range_type)

        # 2. 查询每个数据项在每个日期的可用性(查询12:00点是否存在)
        availability_matrix = []

        for data_item_id in range(1, 16):  # 15个数据项
            row = []
            config = DATA_ITEM_CONFIG[data_item_id]
            collection = self.db[config["collection"]]

            for date_str in date_list:
                # 查询该日期12:00点的数据
                target_datetime = datetime.strptime(date_str, "%Y-%m-%d").replace(
                    hour=12, minute=0, second=0, microsecond=0
                )

                query = {
                    "datetime": target_datetime,
                    **config["filter"]
                }

                doc = collection.find_one(query, {"_id": 1, "datetime": 1})

                row.append({
                    "data_item_id": data_item_id,
                    "date": date_str,
                    "is_available": doc is not None,
                    "sample_timestamp": doc["datetime"].isoformat() if doc else None
                })

            availability_matrix.append(row)

        return {
            "base_date": base_date.strftime("%Y-%m-%d"),
            "date_range": date_list,
            "availability_matrix": availability_matrix
        }

    def get_curve_data(self, data_item_id: int, date: datetime) -> Dict[str, Any]:
        """
        获取单条曲线的96点数据

        Args:
            data_item_id: 数据项ID (1-15)
            date: 查询日期

        Returns:
            曲线数据(96点)
        """
        if data_item_id not in DATA_ITEM_CONFIG:
            raise ValueError(f"无效的数据项ID: {data_item_id}")

        config = DATA_ITEM_CONFIG[data_item_id]
        collection = self.db[config["collection"]]

        # 构造时间范围(左开右闭: (start_of_day, end_of_day])
        start_of_day = date.replace(hour=0, minute=0, second=0, microsecond=0)
        end_of_day = start_of_day + timedelta(days=1)

        query = {
            "datetime": {"$gt": start_of_day, "$lte": end_of_day},
            **config["filter"]
        }

        # 查询并排序
        cursor = collection.find(query).sort("datetime", 1)
        raw_data = list(cursor)

        # 格式化数据点
        formatted_data = [
            self._format_time_point(point, config["field"], date)
            for point in raw_data
        ]

        return {
            "data_item_id": data_item_id,
            "data_item_name": config["name"],
            "date": date.strftime("%Y-%m-%d"),
            "data": formatted_data,
            "total_points": len(formatted_data),
            "completeness": round(len(formatted_data) / 96 * 100, 2)
        }

    def get_multiple_curves(
        self,
        requests: List[Dict[str, Any]]
    ) -> Dict[str, Any]:
        """
        批量获取多条曲线数据

        Args:
            requests: [{"data_item_id": 1, "date": "2025-01-10"}, ...]

        Returns:
            多条曲线数据
        """
        curves = []

        for req in requests:
            data_item_id = req["data_item_id"]
            date = datetime.strptime(req["date"], "%Y-%m-%d")

            try:
                curve_data = self.get_curve_data(data_item_id, date)
                curves.append(curve_data)
            except Exception as e:
                logger.error(f"获取曲线失败: {data_item_id}, {req['date']}: {e}", exc_info=True)
                # 继续处理其他曲线,不中断

        return {"curves": curves}

    # ========== 私有辅助方法 ==========

    def _generate_date_list(self, base_date: datetime, range_type: str) -> List[str]:
        """生成日期列表"""
        if range_type == "recent_3":
            days = [base_date - timedelta(days=i) for i in range(2, -1, -1)]  # D-2, D-1, D
        elif range_type == "recent_7":
            days = [base_date - timedelta(days=i) for i in range(7, -1, -1)]  # D-7到D
        elif range_type == "historical_10":
            days = [base_date - timedelta(days=i) for i in range(10, 0, -1)]  # D-10到D-1
        else:
            raise ValueError(f"无效的日期范围类型: {range_type}")

        return [d.strftime("%Y-%m-%d") for d in days]

    def _format_time_point(
        self,
        point: Dict[str, Any],
        field_name: str,
        query_date: datetime
    ) -> Dict[str, Any]:
        """
        格式化时间点为前端所需格式

        重要: 将次日00:00格式化为"24:00"
        """
        timestamp: datetime = point["datetime"]
        query_next_day = query_date.replace(hour=0, minute=0, second=0, microsecond=0) + timedelta(days=1)

        # 判断是否为某一天的第96个点(次日00:00)
        if timestamp.hour == 0 and timestamp.minute == 0 and timestamp.date() == query_next_day.date():
            time_label = "24:00"
        else:
            time_label = timestamp.strftime("%H:%M")

        return {
            "time": time_label,
            "value": round(point[field_name], 2),
            "timestamp": timestamp.isoformat()
        }
```

```typescript
  try {
    const response = await apiClient.get('/api/v1/forecast-base-data/availability', {
      params: {
        base_date: format(date, 'yyyy-MM-dd'),
        date_range: dateRangeType
      }
    });
    setAvailabilityData(response.data);
  } catch (err: any) {
    setError(err.response?.data?.detail || '加载数据可用性失败');
  } finally {
    setLoading(false);
  }
};

// 加载曲线数据
const fetchCurves = async () => {
  const selectedKeys = Object.keys(selectedItems).filter(key => selectedItems[key]);

  if (selectedKeys.length === 0) {
    setCurvesData([]);
    return;
  }

  setLoadingCurves(true);

  try {
    const requests = selectedKeys.map(key => {
      const [data_item_id, date] = key.split('_');
      return { data_item_id: parseInt(data_item_id), date };
    });

    const response = await apiClient.post('/api/v1/forecast-base-data/curves', requests);
    setCurvesData(response.data.curves);
  } catch (err: any) {
    console.error('加载曲线数据失败:', err);
  } finally {
    setLoadingCurves(false);
  }
};

// 自动加载
useEffect(() => {
  fetchAvailability(selectedDate);
}, [selectedDate, dateRangeType]);

useEffect(() => {
  fetchCurves();
}, [selectedItems]);
```

```typescript
// 合并多条曲线到同一时间轴
const prepareChartData = () => {
  if (curvesData.length === 0) return [];

  const timeMap: Record<string, any> = {};

  curvesData.forEach((curve, idx) => {
    curve.data.forEach(point => {
      if (!timeMap[point.time]) {
        timeMap[point.time] = { time: point.time };
      }
      timeMap[point.time][`curve_${idx}`] = point.value;
    });
  });

  return Object.values(timeMap).sort((a, b) => {
    const timeA = a.time === '24:00' ? '23:59:59' : a.time;
    const timeB = b.time === '24:00' ? '23:59:59' : b.time;
    return timeA.localeCompare(timeB);
  });
};
```

```python
def _format_time_point(self, point, field_name, query_date):
    timestamp: datetime = point["datetime"]
    query_next_day = query_date + timedelta(days=1)

    # 关键判断: 次日00:00 → "24:00"
    if timestamp.hour == 0 and timestamp.minute == 0 and timestamp.date() == query_next_day.date():
        time_label = "24:00"
    else:
        time_label = timestamp.strftime("%H:%M")

    return {"time": time_label, "value": point[field_name], "timestamp": timestamp.isoformat()}
```

### C. 常见问题 (FAQ)

#### Q1: 为什么使用12:00点来判断数据可用性?
**A**: 12:00是一天的中间时刻,如果该时刻有数据,说明该天的数据大概率完整。这是业务需求文档中明确的设计决策。

#### Q2: 如何处理数据不完整的情况(少于96点)?
**A**:
- 后端返回实际点数和 `completeness` 字段
- 前端正常渲染曲线,Recharts会自动处理缺失点
- 可选: 在图表上方显示数据完整性指示器

#### Q3: 为什么不在前端进行时间转换?
**A**:
- 遵循项目规范: 时序数据处理职责在后端
- 降低前端复杂度,避免24:00处理逻辑泄漏
- 保持数据格式统一,便于维护

#### Q4: 如何优化数据可用性查询性能?
**A**:
- 使用 `$in` 操作符批量查询
- 只查询 `_id` 和 `datetime` 字段,减少数据传输
- 考虑添加前端缓存 (如果需要)

#### Q5: 移动端表格太挤怎么办?
**A**:
- 使用缩写名称
- 减小字体和内边距
- 支持横向滚动
- 分组显示 (通过Tab切换)
