```python
from pydantic import BaseModel, Field
from typing import List, Optional, Literal
from datetime import datetime


class PricePredictionRequest(BaseModel):
    """价格预测请求模型"""
    date: str = Field(..., description="预测日期，格式 YYYY-MM-DD")
    prediction_type: Literal["day_ahead", "real_time"] = Field(
        default="day_ahead",
        description="预测类型：日前或实时"
    )


class PricePoint(BaseModel):
    """单个价格点"""
    time: str = Field(..., description="时刻点，格式 HH:MM")
    predicted_price: float = Field(..., description="预测价格（元/MWh）")
    confidence_interval_lower: Optional[float] = Field(
        None,
        description="置信区间下界（可选）"
    )
    confidence_interval_upper: Optional[float] = Field(
        None,
        description="置信区间上界（可选）"
    )


class PricePredictionResponse(BaseModel):
    """价格预测响应模型"""
    date: str = Field(..., description="预测日期")
    prediction_type: str = Field(..., description="预测类型")
    predicted_at: datetime = Field(..., description="预测生成时间")
    model_version: str = Field(..., description="模型版本")
    data: List[PricePoint] = Field(..., description="96点价格预测数据")
    statistics: dict = Field(
        ...,
        description="统计信息（最小值、最大值、平均值等）"
    )


class ModelInfo(BaseModel):
    """模型信息"""
    model_id: str
    model_type: str
    target_type: str
    trained_at: datetime
    metrics: dict
    feature_count: int
    is_active: bool


class ModelListResponse(BaseModel):
    """模型列表响应"""
    total: int
    items: List[ModelInfo]
```

#### 3.1.2 预测服务层

**文件位置**：`webapp/services/price_prediction_service.py`

```python
from datetime import datetime, timedelta
from typing import List, Dict, Optional
import pandas as pd
import numpy as np
from pathlib import Path

from webapp.ml.price_prediction_model import PricePredictionModel
from webapp.services.price_prediction_dataset_service import PricePredictionDatasetService
from webapp.tools.mongo import DATABASE
from webapp.models.price_prediction import PricePoint

import logging

logger = logging.getLogger(__name__)


class PricePredictionService:
    """价格预测服务"""

    def __init__(self):
        self.db = DATABASE
        self.dataset_service = PricePredictionDatasetService()
        self.model_cache: Dict[str, PricePredictionModel] = {}

    def predict_price(
        self,
        date: str,
        prediction_type: str = "day_ahead"
    ) -> Dict:
        """
        预测指定日期的价格

        Args:
            date: 预测日期，格式 YYYY-MM-DD
            prediction_type: 预测类型（day_ahead 或 real_time）

        Returns:
            预测结果字典
        """
        target_date = datetime.strptime(date, "%Y-%m-%d")

        # 1. 加载模型
        model, model_info = self._load_active_model(prediction_type)

        # 2. 构建特征数据
        features_df = self._build_prediction_features(target_date, prediction_type)

        if features_df.empty:
            raise ValueError(f"无法构建 {date} 的预测特征，请确保基础数据已加载")

        # 3. 执行预测
        predictions = model.predict(features_df)

        # 4. 格式化输出
        result = self._format_prediction_result(
            date=date,
            prediction_type=prediction_type,
            predictions=predictions,
            features_df=features_df,
            model_info=model_info
        )

        # 5. 保存预测结果到数据库（可选）
        self._save_prediction_result(result, prediction_type)

        return result

    def _load_active_model(
        self,
        prediction_type: str
    ) -> tuple[PricePredictionModel, dict]:
        """
        加载当前激活的模型

        Args:
            prediction_type: 预测类型

        Returns:
            (模型实例, 模型信息字典)
        """
        # 从缓存中查找
        cache_key = f"model_{prediction_type}"

        if cache_key in self.model_cache:
            logger.info(f"从缓存加载模型: {cache_key}")
            model = self.model_cache[cache_key]
        else:
            # 从数据库查询最新的活跃模型
            model_info = self.db["model_training_logs"].find_one(
                {
                    "target_type": prediction_type,
                    "is_active": True
                },
                sort=[("trained_at", -1)]
            )

            if not model_info:
                raise ValueError(f"未找到 {prediction_type} 类型的活跃模型")

            model_path = model_info["model_path"]

            if not Path(model_path).exists():
                raise FileNotFoundError(f"模型文件不存在: {model_path}")

            model = PricePredictionModel.load(model_path)

            # 缓存模型
            self.model_cache[cache_key] = model

            logger.info(f"从文件加载模型: {model_path}")

        # 获取模型信息（用于响应）
        model_info = self.db["model_training_logs"].find_one(
            {
                "target_type": prediction_type,
                "is_active": True
            },
            sort=[("trained_at", -1)]
        )

        return model, model_info

    def _build_prediction_features(
        self,
        target_date: datetime,
        prediction_type: str
    ) -> pd.DataFrame:
        """
        构建预测特征数据

        Args:
            target_date: 目标日期
            prediction_type: 预测类型

        Returns:
            特征 DataFrame（96行，每行对应一个时刻点）
        """
        # 预测 D 日价格时，需要使用 D-1 日及之前的数据
        # 因此查询范围为 [D-1, D]

        start = target_date
        end = target_date + timedelta(days=1)

        # 加载日前发布数据（包含 D 日的预测负荷、新能源等）
        daily_release = self._load_daily_release_for_prediction(start, end)

        # 加载 T-1 日的实际数据（作为滞后特征）
        start_lagged = start - timedelta(days=1)
        generation_lagged = self._load_generation_data_for_prediction(
            start_lagged,
            start
        )

        # 合并特征
        features_df = daily_release.merge(
            generation_lagged,
            on="datetime",
            how="left",
            suffixes=("", "_t1")
        )

        # 特征工程（与训练时保持一致）
        features_df = self.dataset_service._feature_engineering(features_df)

        # 筛选目标日期的数据（96点）
        features_df = features_df[
            (features_df["datetime"] >= target_date) &
            (features_df["datetime"] < target_date + timedelta(days=1))
        ]

        return features_df

    def _load_daily_release_for_prediction(
        self,
        start: datetime,
        end: datetime
    ) -> pd.DataFrame:
        """加载日前发布数据用于预测"""
        cursor = self.db["daily_release"].find(
            {"datetime": {"$gte": start, "$lt": end}},
            {
                "datetime": 1,
                "system_load_forecast": 1,
                "pv_forecast": 1,
                "wind_forecast": 1,
                "tieline_plan": 1,
                "nonmarket_unit_forecast": 1,
                "_id": 0
            }
        ).sort("datetime", 1)

        return pd.DataFrame(list(cursor))

    def _load_generation_data_for_prediction(
        self,
        start: datetime,
        end: datetime
    ) -> pd.DataFrame:
        """加载实时发电出力数据用于预测（滞后特征）"""
        cursor = self.db["real_time_generation"].find(
            {"datetime": {"$gte": start, "$lt": end}},
            {
                "datetime": 1,
                "total_generation": 1,
                "thermal_generation": 1,
                "hydro_generation": 1,
                "wind_generation": 1,
                "solar_generation": 1,
                "renewable_total_generation": 1,
                "_id": 0
            }
        ).sort("datetime", 1)

        df = pd.DataFrame(list(cursor))

        # 将时间戳向后平移1天（T-1 -> T）
        if not df.empty:
            df["datetime"] = df["datetime"] + timedelta(days=1)

        return df

    def _format_prediction_result(
        self,
        date: str,
        prediction_type: str,
        predictions: np.ndarray,
        features_df: pd.DataFrame,
        model_info: dict
    ) -> Dict:
        """
        格式化预测结果

        Args:
            date: 预测日期
            prediction_type: 预测类型
            predictions: 预测价格数组（96个值）
            features_df: 特征数据（包含时间戳）
            model_info: 模型信息

        Returns:
            格式化的预测结果字典
        """
        # 构建 96 点数据列表
        data_points = []

        for i, (idx, row) in enumerate(features_df.iterrows()):
            # 格式化时间标签（业务时间）
            hour = row["hour"]
            minute = row["minute"]

            # 处理 24:00 情况（第96个点）
            if hour == 0 and minute == 0 and i == 95:
                time_str = "24:00"
            else:
                time_str = f"{hour:02d}:{minute:02d}"

            data_points.append(
                PricePoint(
                    time=time_str,
                    predicted_price=round(float(predictions[i]), 2)
                ).dict()
            )

        # 统计信息
        statistics = {
            "min_price": round(float(predictions.min()), 2),
            "max_price": round(float(predictions.max()), 2),
            "avg_price": round(float(predictions.mean()), 2),
            "std_price": round(float(predictions.std()), 2)
        }

        result = {
            "date": date,
            "prediction_type": prediction_type,
            "predicted_at": datetime.now(),
            "model_version": str(model_info["_id"]),
            "data": data_points,
            "statistics": statistics
        }

        return result

    def _save_prediction_result(self, result: Dict, prediction_type: str):
        """
        保存预测结果到数据库（用于历史记录和回测）

        Args:
            result: 预测结果字典
            prediction_type: 预测类型
        """
        collection_name = f"price_predictions_{prediction_type}"

        # 先删除同一日期的旧预测（如果存在）
        self.db[collection_name].delete_many({"date": result["date"]})

        # 插入新预测
        self.db[collection_name].insert_one(result)

        logger.info(f"预测结果已保存到 {collection_name}: {result['date']}")

    def get_model_list(self, prediction_type: Optional[str] = None) -> List[Dict]:
        """
        获取模型列表

        Args:
            prediction_type: 预测类型过滤（可选）

        Returns:
            模型信息列表
        """
        query = {}
        if prediction_type:
            query["target_type"] = prediction_type

        cursor = self.db["model_training_logs"].find(query).sort("trained_at", -1)

        models = []
        for doc in cursor:
            models.append({
                "model_id": str(doc["_id"]),
                "model_type": doc.get("model_type", "Unknown"),
                "target_type": doc["target_type"],
                "trained_at": doc["trained_at"],
                "metrics": doc.get("metrics", {}),
                "feature_count": doc.get("feature_count", 0),
                "is_active": doc.get("is_active", False)
            })

        return models

    def activate_model(self, model_id: str) -> bool:
        """
        激活指定模型（将其他模型设为非激活）

        Args:
            model_id: 模型ID

        Returns:
            是否成功
        """
        from bson import ObjectId

        # 查询模型
        model = self.db["model_training_logs"].find_one({"_id": ObjectId(model_id)})

        if not model:
            raise ValueError(f"模型不存在: {model_id}")

        target_type = model["target_type"]

        # 将同类型的其他模型设为非激活
        self.db["model_training_logs"].update_many(
            {"target_type": target_type},
            {"$set": {"is_active": False}}
        )

        # 激活指定模型
        self.db["model_training_logs"].update_one(
            {"_id": ObjectId(model_id)},
            {"$set": {"is_active": True}}
        )

        # 清除缓存
        cache_key = f"model_{target_type}"
        if cache_key in self.model_cache:
            del self.model_cache[cache_key]

        logger.info(f"模型已激活: {model_id}")

        return True
```

```python
from fastapi import APIRouter, HTTPException, Depends, Query
from datetime import datetime
from typing import Optional

from webapp.services.price_prediction_service import PricePredictionService
from webapp.models.price_prediction import (
    PricePredictionRequest,
    PricePredictionResponse,
    ModelListResponse
)
from webapp.api.auth import get_current_active_user

import logging

logger = logging.getLogger(__name__)

router = APIRouter()


@router.post(
    "/api/v1/price-prediction/predict",
    response_model=PricePredictionResponse,
    dependencies=[Depends(get_current_active_user)]
)
async def predict_price(request: PricePredictionRequest):
    """
    预测指定日期的现货价格（96点）

    Args:
        request: 预测请求（包含日期和预测类型）

    Returns:
        96点价格预测结果
    """
    try:
        service = PricePredictionService()

        result = service.predict_price(
            date=request.date,
            prediction_type=request.prediction_type
        )

        return result

    except ValueError as e:
        logger.error(f"预测失败: {str(e)}", exc_info=True)
        raise HTTPException(status_code=400, detail=str(e))

    except Exception as e:
        logger.error(f"预测失败（系统错误）: {str(e)}", exc_info=True)
        raise HTTPException(status_code=500, detail="预测失败，请稍后重试")


@router.get(
    "/api/v1/price-prediction/models",
    response_model=ModelListResponse,
    dependencies=[Depends(get_current_active_user)]
)
async def get_models(
    prediction_type: Optional[str] = Query(
        None,
        description="预测类型过滤（day_ahead 或 real_time）"
    )
):
    """
    获取模型列表

    Args:
        prediction_type: 预测类型过滤（可选）

    Returns:
        模型列表
    """
    try:
        service = PricePredictionService()

        models = service.get_model_list(prediction_type=prediction_type)

        return {"total": len(models), "items": models}

    except Exception as e:
        logger.error(f"获取模型列表失败: {str(e)}", exc_info=True)
        raise HTTPException(status_code=500, detail="获取模型列表失败")


@router.post(
    "/api/v1/price-prediction/models/{model_id}/activate",
    dependencies=[Depends(get_current_active_user)]
)
async def activate_model(model_id: str):
    """
    激活指定模型

    Args:
        model_id: 模型ID

    Returns:
        成功消息
    """
    try:
        service = PricePredictionService()

        service.activate_model(model_id)

        return {"message": f"模型 {model_id} 已激活"}

    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))

    except Exception as e:
        logger.error(f"激活模型失败: {str(e)}", exc_info=True)
        raise HTTPException(status_code=500, detail="激活模型失败")
```

```python
"""
为价格预测功能创建数据库索引

用法:
    python -m webapp.scripts.create_prediction_indexes

from webapp.tools.mongo import DATABASE

import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


def create_indexes():
    """创建索引"""

    # 1. daily_release 集合（如果不存在）
    DATABASE["daily_release"].create_index([("datetime", 1)], unique=True)
    logger.info("创建索引: daily_release.datetime")

    # 2. real_time_generation 集合（如果不存在）
    DATABASE["real_time_generation"].create_index([("datetime", 1)], unique=True)
    DATABASE["real_time_generation"].create_index([("date_str", 1), ("time_str", 1)])
    logger.info("创建索引: real_time_generation.datetime, date_str+time_str")

    # 3. model_training_logs 集合
    DATABASE["model_training_logs"].create_index([
        ("target_type", 1),
        ("is_active", 1),
        ("trained_at", -1)
    ])
    logger.info("创建索引: model_training_logs.target_type+is_active+trained_at")

    # 4. price_predictions_day_ahead 集合
    DATABASE["price_predictions_day_ahead"].create_index([("date", 1)], unique=True)
    DATABASE["price_predictions_day_ahead"].create_index([("predicted_at", -1)])
    logger.info("创建索引: price_predictions_day_ahead.date, predicted_at")

    # 5. price_predictions_real_time 集合
    DATABASE["price_predictions_real_time"].create_index([("date", 1)], unique=True)
    DATABASE["price_predictions_real_time"].create_index([("predicted_at", -1)])
    logger.info("创建索引: price_predictions_real_time.date, predicted_at")

    logger.info("所有索引创建完成！")


if __name__ == "__main__":
```

```typescript
import apiClient from './client';

export interface PricePoint {
  time: string;
  predicted_price: number;
  confidence_interval_lower?: number;
  confidence_interval_upper?: number;
}

export interface PricePredictionResponse {
  date: string;
  prediction_type: string;
  predicted_at: string;
  model_version: string;
  data: PricePoint[];
  statistics: {
    min_price: number;
    max_price: number;
    avg_price: number;
    std_price: number;
  };
}

export interface ModelInfo {
  model_id: string;
  model_type: string;
  target_type: string;
  trained_at: string;
  metrics: {
    mae: number;
    rmse: number;
    mape: number;
    r2: number;
  };
  feature_count: number;
  is_active: boolean;
}

/**
 * 预测价格
 */
export const predictPrice = async (
  date: string,
  predictionType: 'day_ahead' | 'real_time' = 'day_ahead'
): Promise<PricePredictionResponse> => {
  const response = await apiClient.post('/api/v1/price-prediction/predict', {

    prediction_type: predictionType,
  });

};

/**
 * 获取模型列表

  return response.data;
};

/**
 * 激活模型
 */
export const activateModel = async (modelId: string): Promise<void> => {
  await apiClient.post(`/api/v1/price-prediction/models/${modelId}/activate`);
```

```typescript
import React, { useState, useRef } from 'react';
import {
  Box,
  Paper,
  Typography,
  IconButton,
  Alert,
  CircularProgress,
  Grid,
  Button,
} from '@mui/material';
import { DatePicker, LocalizationProvider } from '@mui/x-date-pickers';
import { AdapterDateFns } from '@mui/x-date-pickers/AdapterDateFns';
import { zhCN } from 'date-fns/locale';
import ArrowLeftIcon from '@mui/icons-material/ArrowLeft';
import ArrowRightIcon from '@mui/icons-material/ArrowRight';
import { format, addDays } from 'date-fns';
import { ResponsiveContainer, LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, Legend } from 'recharts';

import { predictPrice, PricePredictionResponse } from '../api/pricePrediction';
import { useChartFullscreen } from '../hooks/useChartFullscreen';

export const PricePredictionPage: React.FC = () => {
  const [selectedDate, setSelectedDate] = useState<Date | null>(addDays(new Date(), 1)); // 默认预测明天
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [predictionData, setPredictionData] = useState<PricePredictionResponse | null>(null);

  const chartRef = useRef<HTMLDivElement>(null);
  const dateStr = selectedDate ? format(selectedDate, 'yyyy-MM-dd') : '';

  const { isFullscreen, FullscreenEnterButton, FullscreenExitButton, FullscreenTitle, NavigationButtons } =
    useChartFullscreen({
      chartRef,
      title: `价格预测 (${dateStr})`,
      onPrevious: () => handleShiftDate(-1),
      onNext: () => handleShiftDate(1),
    });

  const fetchPrediction = (date: Date | null) => {
    if (!date) return;
    setLoading(true);
    setError(null);

    const formattedDate = format(date, 'yyyy-MM-dd');

    predictPrice(formattedDate, 'day_ahead')
      .then((response) => {
        setPredictionData(response);
      })
      .catch((error) => {
        console.error('Error:', error);
        setError(error.response?.data?.detail || error.message || '预测失败');
        setPredictionData(null);
      })
      .finally(() => setLoading(false));
  };

  const handleShiftDate = (days: number) => {
    if (!selectedDate) return;
    const newDate = addDays(selectedDate, days);
    setSelectedDate(newDate);
  };

  React.useEffect(() => {
    fetchPrediction(selectedDate);
  }, [selectedDate]);

  // 转换数据格式用于图表显示
  const chartData = predictionData
    ? predictionData.data.map((point) => ({
        time: point.time,
        price: point.predicted_price,
      }))
    : [];

  return (
    <LocalizationProvider dateAdapter={AdapterDateFns} adapterLocale={zhCN}>
      <Box>
        {/* 日期选择器 */}
        <Paper variant="outlined" sx={{ p: 2, display: 'flex', gap: 1, alignItems: 'center', flexWrap: 'wrap' }}>
          <IconButton onClick={() => handleShiftDate(-1)} disabled={loading}>
            <ArrowLeftIcon />
          </IconButton>
          <DatePicker
            label="选择预测日期"
            value={selectedDate}
            onChange={(date) => setSelectedDate(date)}
            disabled={loading}
            slotProps={{ textField: { sx: { width: { xs: '150px', sm: '200px' } } } }}
          />
          <IconButton onClick={() => handleShiftDate(1)} disabled={loading}>
            <ArrowRightIcon />
          </IconButton>
        </Paper>

        {/* Loading 状态 */}
        {loading && !predictionData ? (
          <Box display="flex" justifyContent="center" alignItems="center" minHeight="400px">
            <CircularProgress />
          </Box>
        ) : error ? (
          <Alert severity="error" sx={{ mt: 2 }}>
            {error}
          </Alert>
        ) : predictionData ? (
          <Box sx={{ position: 'relative' }}>
            {/* 数据刷新覆盖层 */}
            {loading && (
              <Box
                sx={{
                  position: 'absolute',
                  top: 0,
                  left: 0,
                  right: 0,
                  bottom: 0,
                  display: 'flex',
                  alignItems: 'center',
                  justifyContent: 'center',
                  backgroundColor: 'rgba(255, 255, 255, 0.7)',
                  zIndex: 1000,
                }}
              >
                <CircularProgress />
              </Box>
            )}

            {/* 统计信息 */}
            <Paper variant="outlined" sx={{ p: 2, mt: 2 }}>
              <Typography variant="h6" gutterBottom>
                预测统计信息
              </Typography>
              <Grid container spacing={{ xs: 1, sm: 2 }}>
                <Grid size={{ xs: 6, md: 3 }}>
                  <Typography variant="body2" color="text.secondary">
                    最低价
                  </Typography>
                  <Typography variant="h6">{predictionData.statistics.min_price.toFixed(2)} 元/MWh</Typography>
                </Grid>
                <Grid size={{ xs: 6, md: 3 }}>
                  <Typography variant="body2" color="text.secondary">
                    最高价
                  </Typography>
                  <Typography variant="h6">{predictionData.statistics.max_price.toFixed(2)} 元/MWh</Typography>
                </Grid>
                <Grid size={{ xs: 6, md: 3 }}>
                  <Typography variant="body2" color="text.secondary">
                    平均价
                  </Typography>
                  <Typography variant="h6">{predictionData.statistics.avg_price.toFixed(2)} 元/MWh</Typography>
                </Grid>
                <Grid size={{ xs: 6, md: 3 }}>
                  <Typography variant="body2" color="text.secondary">
                    标准差
                  </Typography>
                  <Typography variant="h6">{predictionData.statistics.std_price.toFixed(2)}</Typography>
                </Grid>
              </Grid>
            </Paper>

            {/* 价格曲线图 */}
            <Paper variant="outlined" sx={{ p: 2, mt: 2 }}>
              <Typography variant="h6" gutterBottom>
                96点价格预测曲线
              </Typography>
              <Box
                ref={chartRef}
                sx={{
                  height: { xs: 350, sm: 400 },
                  position: 'relative',
                  backgroundColor: isFullscreen ? 'background.paper' : 'transparent',
                  p: isFullscreen ? 2 : 0,
                  ...(isFullscreen && {
                    position: 'fixed',
                    top: 0,
                    left: 0,
                    width: '100vw',
                    height: '100vh',
                    zIndex: 1400,
                  }),
                }}
              >
                <FullscreenEnterButton />
                <FullscreenExitButton />
                <FullscreenTitle />
                <NavigationButtons />

                {!chartData || chartData.length === 0 ? (
                  <Box sx={{ display: 'flex', height: '100%', alignItems: 'center', justifyContent: 'center' }}>
                    <Typography>无预测数据</Typography>
                  </Box>
                ) : (
                  <ResponsiveContainer width="100%" height="100%">
                    <LineChart data={chartData}>
                      <CartesianGrid strokeDasharray="3 3" />
                      <XAxis dataKey="time" />
                      <YAxis label={{ value: '价格 (元/MWh)', angle: -90, position: 'insideLeft' }} />
                      <Tooltip />
                      <Legend />
                      <Line type="monotone" dataKey="price" stroke="#8884d8" name="预测价格" strokeWidth={2} />
                    </LineChart>
                  </ResponsiveContainer>
                )}
              </Box>
            </Paper>
          </Box>
        ) : null}
      </Box>
    </LocalizationProvider>
  );
};
```
