## 阶段4：测试与优化（1周）

### 4.1 回测验证

**目标**：使用历史数据验证模型实际表现

**步骤**：

1. **准备回测数据集**
   ```python
   # 文件位置：webapp/scripts/backtest_model.py

   from datetime import datetime, timedelta
   from webapp.services.price_prediction_service import PricePredictionService
   from webapp.tools.mongo import DATABASE
   import pandas as pd
   import numpy as np

   def backtest_model(start_date: str, end_date: str):
       """
       回测模型预测效果

       Args:
           start_date: 回测起始日期
           end_date: 回测结束日期
       """
       service = PricePredictionService()

       start = datetime.strptime(start_date, "%Y-%m-%d")
       end = datetime.strptime(end_date, "%Y-%m-%d")

       results = []

       current_date = start
       while current_date <= end:
           date_str = current_date.strftime("%Y-%m-%d")

           try:
               # 1. 生成预测
               prediction = service.predict_price(date_str, "day_ahead")

               # 2. 获取实际价格
               actual_prices = DATABASE["day_ahead_spot_price"].find({
                   "date_str": date_str
               }).sort("datetime", 1)

               actual_prices_list = [doc["avg_clearing_price"] for doc in actual_prices]

               # 3. 计算误差
               predicted_prices = [p["predicted_price"] for p in prediction["data"]]

               if len(actual_prices_list) == len(predicted_prices):
                   mae = np.mean(np.abs(np.array(actual_prices_list) - np.array(predicted_prices)))
                   mape = np.mean(np.abs((np.array(actual_prices_list) - np.array(predicted_prices)) / np.array(actual_prices_list))) * 100

                   results.append({
                       "date": date_str,
                       "mae": mae,
                       "mape": mape
                   })

                   print(f"{date_str}: MAE={mae:.2f}, MAPE={mape:.2f}%")

           except Exception as e:
               print(f"{date_str}: 回测失败 - {str(e)}")

           current_date += timedelta(days=1)

       # 4. 汇总结果
       df = pd.DataFrame(results)

       print("\n" + "="*60)
       print("回测结果汇总:")
       print("="*60)
       print(f"平均 MAE: {df['mae'].mean():.2f} 元/MWh")
       print(f"平均 MAPE: {df['mape'].mean():.2f}%")
       print(f"MAE 标准差: {df['mae'].std():.2f}")
       print(f"MAPE 标准差: {df['mape'].std():.2f}")
       print("="*60)

       return df

   if __name__ == "__main__":
       # 回测最近30天
       end_date = (datetime.now() - timedelta(days=1)).strftime("%Y-%m-%d")
       start_date = (datetime.now() - timedelta(days=30)).strftime("%Y-%m-%d")

       backtest_model(start_date, end_date)
   ```

2. **运行回测**
   ```bash
   python -m webapp.scripts.backtest_model
   ```

3. **分析回测结果**
   - 检查平均误差是否在可接受范围
   - 识别误差较大的日期（节假日、极端天气等）
   - 评估模型稳定性（误差标准差）

### 5.2 模型监控与维护

**目标**：持续跟踪模型表现，及时发现性能退化

#### 5.2.1 创建监控服务

**文件位置**：`webapp/services/model_monitoring_service.py`

```python
from datetime import datetime, timedelta
from typing import Dict
from webapp.tools.mongo import DATABASE
import numpy as np

class ModelMonitoringService:
    """模型性能监控服务"""

    def __init__(self):
        self.db = DATABASE

    def calculate_daily_metrics(self, date: str, prediction_type: str = "day_ahead") -> Dict:
        """
        计算指定日期的预测误差指标

        Args:
            date: 日期字符串（YYYY-MM-DD）
            prediction_type: 预测类型

        Returns:
            误差指标字典
        """
        # 1. 获取预测结果
        prediction_collection = f"price_predictions_{prediction_type}"
        prediction = self.db[prediction_collection].find_one({"date": date})

        if not prediction:
            raise ValueError(f"未找到 {date} 的预测结果")

        predicted_prices = [p["predicted_price"] for p in prediction["data"]]

        # 2. 获取实际价格
        actual_collection = f"{prediction_type}_spot_price"
        actual_cursor = self.db[actual_collection].find({
            "date_str": date
        }).sort("datetime", 1)

        actual_prices = [doc["avg_clearing_price"] for doc in actual_cursor]

        if len(actual_prices) != len(predicted_prices):
            raise ValueError(f"预测点数与实际点数不匹配: {len(predicted_prices)} vs {len(actual_prices)}")

        # 3. 计算误差
        predicted = np.array(predicted_prices)
        actual = np.array(actual_prices)

        mae = np.mean(np.abs(predicted - actual))
        rmse = np.sqrt(np.mean((predicted - actual) ** 2))
        mape = np.mean(np.abs((actual - predicted) / actual)) * 100

        metrics = {
            "date": date,
            "prediction_type": prediction_type,
            "mae": float(mae),
            "rmse": float(rmse),
            "mape": float(mape),
            "calculated_at": datetime.now()
        }

        # 4. 保存到监控集合
        self.db["model_performance_metrics"].insert_one(metrics)

        return metrics

    def check_model_drift(self, days: int = 7) -> Dict:
        """
        检查模型性能退化（过去N天的平均误差）

        Args:
            days: 回溯天数

        Returns:
            性能退化报告
        """
        end_date = datetime.now()
        start_date = end_date - timedelta(days=days)

        cursor = self.db["model_performance_metrics"].find({
            "calculated_at": {"$gte": start_date}
        })

        metrics_list = list(cursor)

        if not metrics_list:
            return {"status": "no_data", "message": "无监控数据"}

        avg_mae = np.mean([m["mae"] for m in metrics_list])
        avg_mape = np.mean([m["mape"] for m in metrics_list])

        # 定义性能阈值
        MAE_THRESHOLD = 50
        MAPE_THRESHOLD = 15

        status = "healthy"
        if avg_mae > MAE_THRESHOLD or avg_mape > MAPE_THRESHOLD:
            status = "degraded"

        return {
            "status": status,
            "avg_mae": avg_mae,
            "avg_mape": avg_mape,
            "days": days,
            "sample_count": len(metrics_list),
            "message": "模型性能正常" if status == "healthy" else "模型性能下降，建议重新训练"
        }
```

#### 5.2.2 创建定时任务

```python
"""
模型监控定时任务

建议使用 Cron 每日执行：
0 10 * * * python -m webapp.tasks.model_monitoring_task
"""

from datetime import datetime, timedelta
from webapp.services.model_monitoring_service import ModelMonitoringService
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


def run_daily_monitoring():
    """每日监控任务"""
    service = ModelMonitoringService()

    # 计算前一天的误差指标
    yesterday = (datetime.now() - timedelta(days=1)).strftime("%Y-%m-%d")

    try:
        metrics = service.calculate_daily_metrics(yesterday, "day_ahead")
        logger.info(f"日期 {yesterday} 的预测误差: MAE={metrics['mae']:.2f}, MAPE={metrics['mape']:.2f}%")

        # 检查模型性能退化
        drift_report = service.check_model_drift(days=7)

        if drift_report["status"] == "degraded":
            logger.warning(f"模型性能下降！过去7天平均 MAE={drift_report['avg_mae']:.2f}, MAPE={drift_report['avg_mape']:.2f}%")
            # TODO: 发送告警邮件/钉钉通知

    except Exception as e:
        logger.error(f"监控任务失败: {str(e)}", exc_info=True)


if __name__ == "__main__":
    run_daily_monitoring()
```

### 5.3 模型更新策略

**推荐策略**：**每周重新训练模型**

**原因**：
- 电力市场价格受季节、天气、政策等因素影响，需要定期更新
- 每周更新频率可以平衡性能和计算成本

**实施步骤**：

1. **创建模型重训脚本**

```bash
# 文件位置：scripts/retrain_model_weekly.sh

#!/bin/bash

# 获取过去6个月的数据作为训练集
END_DATE=$(date +%Y-%m-%d)
START_DATE=$(date -d "180 days ago" +%Y-%m-%d)

echo "开始重新训练模型..."
echo "训练数据范围: $START_DATE 到 $END_DATE"

# 运行训练脚本
python -m webapp.scripts.train_price_model \
    --start-date "$START_DATE" \
    --end-date "$END_DATE" \
    --target-type day_ahead \
    --test-size 0.2

# 检查训练是否成功
if [ $? -eq 0 ]; then
    echo "模型训练成功！"

    # 激活新模型（需要手动确认模型ID）
    # python -m webapp.scripts.activate_latest_model

else
    echo "模型训练失败，请检查日志"
    exit 1
fi
```

2. **配置 Cron（每周日凌晨2点执行）**

```bash
0 2 * * 0 /path/to/scripts/retrain_model_weekly.sh >> /path/to/logs/retrain.log 2>&1
```
