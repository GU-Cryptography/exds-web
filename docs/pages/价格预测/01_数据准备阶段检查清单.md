### 1.2 数据完整性检查

**目标**：识别数据缺失情况，制定填补策略

```python
# 在 webapp 后端创建数据质量检查脚本
# 文件位置：webapp/scripts/data_quality_check.py

from datetime import datetime, timedelta
from webapp.tools.mongo import DATABASE
import pandas as pd

def check_price_data_completeness(start_date: str, end_date: str):
    """
    检查价格数据的完整性

    Args:
        start_date: 起始日期，格式 YYYY-MM-DD
        end_date: 结束日期，格式 YYYY-MM-DD
    """
    start = datetime.strptime(start_date, "%Y-%m-%d")
    end = datetime.strptime(end_date, "%Y-%m-%d")

    # 1. 计算期望的总记录数（每天96个点）
    days = (end - start).days + 1
    expected_records = days * 96

    # 2. 查询实际记录数
    actual_records = DATABASE["day_ahead_spot_price"].count_documents({
        "datetime": {"$gte": start, "$lte": end}
    })

    completeness_rate = (actual_records / expected_records) * 100

    print(f"期望记录数: {expected_records}")
    print(f"实际记录数: {actual_records}")
    print(f"完整率: {completeness_rate:.2f}%")

    # 3. 查找缺失的日期
    pipeline = [
        {"$match": {"datetime": {"$gte": start, "$lte": end}}},
        {
            "$group": {
                "_id": {"$dateToString": {"format": "%Y-%m-%d", "date": "$datetime"}},
                "count": {"$sum": 1}
            }
        },
        {"$sort": {"_id": 1}}
    ]

    daily_counts = list(DATABASE["day_ahead_spot_price"].aggregate(pipeline))

    # 检查哪些日期的数据点不足96个
    incomplete_dates = [item for item in daily_counts if item["count"] < 96]

    if incomplete_dates:
        print(f"\n发现 {len(incomplete_dates)} 个日期的数据不完整:")
        for item in incomplete_dates[:10]:  # 只显示前10个
            print(f"  {item['_id']}: {item['count']}/96 点")

    return {
        "expected": expected_records,
        "actual": actual_records,
        "completeness_rate": completeness_rate,
        "incomplete_dates": incomplete_dates
    }


def check_feature_data_alignment():
    """
    检查特征数据与价格数据的时间对齐情况
    """
    # 获取价格数据的日期范围
    price_dates = DATABASE["day_ahead_spot_price"].distinct("date_str")

    # 检查各特征数据集的覆盖率
    feature_collections = {
        "daily_release": "日前发布数据",
        "real_time_generation": "实时发电出力",
        "real_time_tieline": "实时联络线"
    }

    for collection, name in feature_collections.items():
        feature_dates = set(DATABASE[collection].distinct("date_str"))
        price_dates_set = set(price_dates)

        coverage = len(feature_dates & price_dates_set) / len(price_dates_set) * 100
        missing_dates = price_dates_set - feature_dates

        print(f"\n{name} 覆盖率: {coverage:.2f}%")
        if missing_dates and len(missing_dates) < 10:
            print(f"缺失日期: {sorted(missing_dates)}")


if __name__ == "__main__":
    # 检查最近3个月的数据
```

### 1.3 数据异常值检查

**目标**：识别并处理异常值（如价格为负值、出力超过装机容量等）

```python
# webapp/scripts/outlier_detection.py

def check_price_outliers():
    """检查价格数据的异常值"""
    pipeline = [
        {
            "$group": {
                "_id": None,
                "min_price": {"$min": "$avg_clearing_price"},
                "max_price": {"$max": "$avg_clearing_price"},
                "avg_price": {"$avg": "$avg_clearing_price"},
                "std_price": {"$stdDevPop": "$avg_clearing_price"}
            }
        }
    ]

    stats = list(DATABASE["day_ahead_spot_price"].aggregate(pipeline))[0]

    print(f"价格统计信息:")
    print(f"  最小值: {stats['min_price']:.2f} 元/MWh")
    print(f"  最大值: {stats['max_price']:.2f} 元/MWh")
    print(f"  平均值: {stats['avg_price']:.2f} 元/MWh")
    print(f"  标准差: {stats['std_price']:.2f} 元/MWh")

    # 查找超过3倍标准差的异常值
    threshold_high = stats['avg_price'] + 3 * stats['std_price']
    threshold_low = stats['avg_price'] - 3 * stats['std_price']

    outliers = DATABASE["day_ahead_spot_price"].count_documents({
        "$or": [
            {"avg_clearing_price": {"$gt": threshold_high}},
            {"avg_clearing_price": {"$lt": threshold_low}}
        ]
    })

    print(f"\n异常值数量: {outliers}")

    # 检查负价格
    negative_prices = DATABASE["day_ahead_spot_price"].count_documents({
        "avg_clearing_price": {"$lt": 0}
    })

    print(f"负价格记录数: {negative_prices}")


def check_generation_outliers():
    """检查发电出力数据的异常值"""
    # 检查是否有负出力（除抽蓄外）
    negative_generation = DATABASE["real_time_generation"].count_documents({
        "$or": [
            {"thermal_generation": {"$lt": 0}},
            {"hydro_generation": {"$lt": 0}},
            {"wind_generation": {"$lt": 0}},
            {"solar_generation": {"$lt": 0}}
        ]
    })

    print(f"\n发电出力异常值检查:")
    print(f"负出力记录数: {negative_generation}")
```

## 第二步：特征工程准备（评估）

### 2.1 已有特征列表

基于现有数据集，可以提取的特征包括：

**时间特征**：
- 日期：年、月、日、星期几、是否工作日、是否节假日
- 时刻：小时、15分钟间隔编号（1-96）
- 季节：春夏秋冬

**负荷特征**（从 `daily_release`）：
- `system_load_forecast` - 系统负荷预测值

**新能源特征**（从 `daily_release`、`real_time_generation`）：
- `pv_forecast` / `solar_generation` - 光伏出力
- `wind_forecast` / `wind_generation` - 风电出力

**电源结构特征**（从 `real_time_generation`）：
- `thermal_generation` - 火电出力
- `hydro_generation` - 水电出力
- `pumped_storage_generation` - 抽蓄出力
- `renewable_total_generation` - 新能源总出力

**联络线特征**（从 `daily_release`、`real_time_tieline`）：
- `tieline_plan` / `total_tieline_plan` - 联络线计划

**衍生特征**（需要计算）：
- 净负荷 = 系统负荷 - 新能源出力
- 新能源渗透率 = 新能源出力 / 系统负荷
- 火电裕度 = 火电装机容量 - 火电出力
- 价格滞后项（T-1、T-2、T-96 时刻的价格）
- 移动平均价格（过去7天、30天平均）

### 3.1 推荐技术栈

**数据处理**：
- `pandas` - 数据清洗、转换

- `scikit-learn.metrics` - RMSE, MAE, MAPE
